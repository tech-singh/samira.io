  <script>
    document.querySelectorAll('.reel-container').forEach(container => {
      const video = container.querySelector('.reel-video');
      const playPauseBtn = container.querySelector('.play-pause-btn');
  
      // Toggle play/pause when clicking anywhere on the video
      video.addEventListener('click', () => {
        if (video.paused) {
          video.play();
          playPauseBtn.innerHTML = '❚❚';
        } else {
          video.pause();
          playPauseBtn.innerHTML = '▶';
        }
      });
  
      // Toggle play/pause when clicking the play/pause button
      playPauseBtn.addEventListener('click', (event) => {
        event.stopPropagation(); // Prevent the video click event from firing
        if (video.paused) {
          video.play();
          playPauseBtn.innerHTML = '❚❚';
        } else {
          video.pause();
          playPauseBtn.innerHTML = '▶';
        }
      });
    });
  </script>
